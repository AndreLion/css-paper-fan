  3. Javascript*

    $(function(){
  var renderThread,fanThread;
  $('.fan-trigger').on('mouseover',function(ev){
    if(fanThread){
      clearTimeout(fanThread);
    }
    $('#fan').addClass('show');
    fanThread = setTimeout(function(){
      $('#fan').find('li').addClass('show');
    },10);
  });
  $('#fan').on('mouseleave',function(ev){
    if(fanThread){
      clearTimeout(fanThread);
    }
    $('#fan').removeClass('show');
    fanThread = setTimeout(function(){
      $('#fan').find('li').removeClass('show');
    },10);
  });
});
